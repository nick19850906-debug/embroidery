from fastapi import FastAPI, UploadFile, File
from fastapi.responses import JSONResponse
from fastapi.middleware.cors import CORSMiddleware
import cv2
import numpy as np
from google import genai
from google.genai import types
import traceback

app = FastAPI()

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

client = genai.Client()

# 서버 깨우기용 주소 (여기로 신호를 보내 서버를 기상시킵니다)
@app.get("/")
async def health_check():
    return {"status": "awake"}

def calculate_stitch_count(image_bytes: bytes) -> int:
    nparr = np.frombuffer(image_bytes, np.uint8)
    img = cv2.imdecode(nparr, cv2.IMREAD_GRAYSCALE)
    
    max_width = 600
    height, width = img.shape
    if width > max_width:
        ratio = max_width / width
        new_height = int(height * ratio)
        img = cv2.resize(img, (max_width, new_height), interpolation=cv2.INTER_AREA)
    
    _, binary = cv2.threshold(img, 127, 255, cv2.THRESH_BINARY_INV)
    dist_transform = cv2.distanceTransform(binary, cv2.DIST_L2, 5)
    
    satin_pixels = np.sum((dist_transform > 0) & (dist_transform < 15))
    tatami_pixels = np.sum(dist_transform >= 15)
    
    scale_factor = (width / max_width)**2 if width > max_width else 1.0
    estimated_stitches = int(((satin_pixels * 0.15) + (tatami_pixels * 0.25)) * scale_factor)
    
    return estimated_stitches

@app.post("/api/estimate")
async def estimate_embroidery(file: UploadFile = File(...)):
    try:
        image_bytes = await file.read()
        estimated_stitches = calculate_stitch_count(image_bytes)
        
        prompt = f"""
        당신은 20년 경력의 디지털 자수 전문가입니다. 
        사용자가 업로드한 자수 도안 이미지를 분석하고, 1차 계산된 예상 침수({estimated_stitches}침)를 바탕으로 최종 견적을 내주세요.
        
        다음 형식으로 작성하세요:
        1. 디자인 분석: (복잡도 등)
        2. 추천 자수 기법:
        3. 예상 침수: {estimated_stitches}침
        4. 예상 단가: 1,000침당 $1.50 기준 총비용
        """
        
        response = client.models.generate_content(
            model='gemini-2.5-flash',
            contents=[
                types.Part.from_bytes(data=image_bytes, mime_type=file.content_type),
                prompt
            ]
        )
        
        return {"stitch_count": estimated_stitches, "expert_quote": response.text}
    except Exception as e:
        error_msg = traceback.format_exc()
        print(error_msg) # Render 로그에 에러 상세 원인 기록
        return JSONResponse(status_code=500, content={"expert_quote": f"서버 오류: {str(e)}"})
